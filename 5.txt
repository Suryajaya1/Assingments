    public class SBI
    {
        public virtual float ROI { get; } = 0.102f;
        public float CalcInterest(int p, int t, float r)
        {
            return (p * t * r) / 100;
        }
        public int t(int year, int month)
        {
            return ((12 * year) + month);
        }
        public float Installments(int t, int principle)
        {

            return (principle / t);
        }
    }
    public class SyndicateBank : SBI
    {
        public override float ROI { get; } = 0.9f;
    }
    public class AxisBank : SBI
    {
        public override float ROI { get; } = 0.8f;

    }
    public class CosmosBank : SBI
    {
        public override float ROI { get; } = 0.6f;
    }

    public class MyClass
    {

        {
            SBI rb = new SBI();

            float interest;
            int amt;

            year = Convert.ToInt32(Console.ReadLine());

            month = Convert.ToInt32(Console.ReadLine());

            {
                Console.WriteLine("\nInvalid input..");
                return;
            }

            Console.WriteLine("\n\n\t\t Select the BANK ");

            Console.WriteLine("\n\t 2. Axis Bank  ");
            Console.WriteLine("\n\t 3. Cosmos Bank   ");

    public abstract class EmployeeSalary
    {
        public abstract int TotalSalary();
    }
    interface ICompany
    {
        void Struct();
    }
    public class Employee : EmployeeSalary, ICompany
    {
        public int id;
        public string name;
        public int baseSalary;
        public int Food;
        public int ProvisionalFund;
        public int Transport;
        public int Insurance;
        public override int TotalSalary()
        {

            return ((baseSalary + Food + Transport) - (Insurance + ProvisionalFund));
        }
        public void Struct()
        {
            Console.WriteLine("\t\n Employee Salary Details");
            Console.WriteLine("\nEmployee Name: " + name);
            Console.WriteLine("\n\nYour Total Salary is : " + TotalSalary());
        }
    }

    public class Program
    {
        public static void Main(string[] args)

    {
        public override void Square(int n1)
        {
            Console.WriteLine(n1 * n1);
        }
        public override void Cube(int n1)
        {
            Console.WriteLine(n1 * n1 * n1);

        }
        public override void Factorial(int n1)
        {
            int Fact = 1;
            for (int i = 1; i < n1+1; i++)
            {
                Fact = Fact * i;
            }
            Console.WriteLine(Fact);
            Console.ReadLine();
        }

    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.Write(" Enter the Number     :");
            int n1 = Convert.ToInt32(Console.ReadLine());

            Console.Write(" Square of the Number :");
            s1.Square(n1);
            Console.Write(" Cube of the Number   :");
